+def normalize_cik(raw: str) -> str:
+    """Normalize a Coverage Manager CIK to a zero-padded 10-digit string.
+
+    EDGAR APIs want a 10-digit zero-padded CIK. CM stores it variously
+    (with/without leading zeros, sometimes a 'CIK' prefix). Empty -> ''.

+def derive_filer_type(row: dict) -> str:
+    """domestic (10-K filer) vs foreign (20-F/IFRS filer).

+    The forensic rubric is built on 10-K disclosures (financial-statement notes,
+    MD&A). ADR / cross-listed foreign private issuers file a 20-F with different
+    structure and often IFRS, which the 10-K-based rubric cannot evaluate
+    cleanly — so they must be routed to manual review (Data Gap tier), NOT
+    silently screened as Green. We TAG rather than drop so coverage stays visible.

+def build_new_watchlist(

+    existing: dict[str, dict],

+def load_sec_ticker_ciks() -> dict[str, str]:
+    """Return {TICKER: zero-padded-10-digit CIK} from SEC's ticker map.
+
+    Cached at SEC_TICKERS_CACHE; fetched once (free, no key — just a descriptive
+    User-Agent per SEC policy). On any fetch failure with no cache, returns {} and
+    the extras step degrades loudly (names without a CIK can't be EDGAR-screened).
+    """
+    data = None
+    if SEC_TICKERS_CACHE.exists():
+        try:
+            data = json.loads(SEC_TICKERS_CACHE.read_text(encoding="utf-8"))
+        except Exception:
+            data = None
+    if data is None:
+        ua = os.environ.get("EDGAR_IDENTITY") or "Jason Peterson jroypeterson@gmail.com"
+        req = urllib.request.Request(SEC_TICKERS_URL, headers={"User-Agent": ua})
+        try:
+            with urllib.request.urlopen(req, timeout=30) as resp:
+                data = json.loads(resp.read().decode("utf-8"))
+            SEC_TICKERS_CACHE.write_text(json.dumps(data), encoding="utf-8")
+        except Exception as e:
+            print(f"WARNING: could not fetch SEC ticker map ({e}); "
+                  f"S&P 500 extras without a cached CIK will be skipped.", file=sys.stderr)
+            return {}
+    out: dict[str, str] = {}
+    for rec in data.values():
+        t = str(rec.get("ticker", "")).strip().upper()
+        cik = rec.get("cik_str")
+        if t and cik is not None:
+            out[t] = str(cik).zfill(10)
+    return out

   115	
   116	    for r in watchlist:
   117	        tk = (r.get("ticker") or "").strip().upper()
   118	        if not tk:
   119	            continue
   120	        c = _cohort(r)
   121	        b = coh[c]
   122	        b["total"] += 1
   123	        is_foreign = r.get("filer_type") == "foreign"
   124	        if is_foreign:
   125	            b["foreign"] += 1
   126	        else:
   127	            b["domestic"] += 1
   128	        sc = screens.get(tk)
   129	        if sc:
   130	            b["screened"] += 1
   131	            tier = sc["tier"] if sc["tier"] in b["tiers"] else "Green"
   132	            b["tiers"][tier] = b["tiers"].get(tier, 0) + 1
   133	            if sc["tier"] in ACTIONABLE:
   134	                b["flagged"].append({
   135	                    "ticker": tk, "tier": sc["tier"],
   136	                    "concern": _first_concern(sc["flag_details"]),
   137	                })
   138	        elif not is_foreign:
   139	            b["pending"] += 1
   140	
   141	    # Disjoint roster sizes = what SHOULD be in each ring, INCLUDING names forensic
   142	    # can't screen (biopharma is excluded at sync; foreign is Data-Gap). Surfacing
   143	    # `excluded` keeps the denominator honest — a hidden 10-name gap on Portfolio
   144	    # reads as "covered" otherwise.
   145	    roster_size = _load_cohort_totals()
   146	    if roster_size is None:  # no baked totals -> compute live from rosters
   147	        p, rr, co, sp = (rosters["portfolio"], rosters["researching"],
   148	                         rosters["core"], rosters["sp500"])
   149	        roster_size = {
   150	            "portfolio": len(p),
   151	            "researching": len(rr - p),
   152	            "core": len(co - rr - p),
   153	            "sp500": len(sp - co - rr - p),
   154	        }
   155	    roster_size = dict(roster_size)
   156	    roster_size["other"] = coh["other"]["total"]
   157	    for c in COHORTS:
   158	        rsize = roster_size.get(c, coh[c]["total"])
   159	        coh[c]["roster"] = rsize
   160	        in_wl = coh[c]["domestic"] + coh[c]["foreign"]
   161	        coh[c]["excluded"] = max(0, rsize - in_wl)
   162	
   163	    # Per-ring ETA. Screening runs top-down by priority, so a ring is only complete
   164	    # once everything ABOVE it plus its own pending is screened -> cumulative days.
   165	    # A ring with 0 pending is already complete now (regardless of rings above it).
   166	    cum_pending = 0
   167	    for c in COHORTS:
   168	        b = coh[c]
   169	        cum_pending += b["pending"]
   170	        if b["pending"] == 0:
   171	            b["eta_days"] = 0
   172	            b["eta_date"] = None  # already complete
   173	        elif per_day > 0:
   174	            days = -(-cum_pending // per_day)  # ceil of cumulative pending
   175	            b["eta_days"] = days
   176	            b["eta_date"] = (date.fromisoformat(today) + timedelta(days=days)).isoformat()
   177	        else:
   178	            b["eta_days"] = None
   179	            b["eta_date"] = None
   180	
   181	    # Overall roll-up (domestic denominators — foreign are Data-Gap, not screenable).
   182	    dom_total = sum(b["domestic"] for b in coh.values())
   183	    dom_screened = sum(b["screened"] for b in coh.values())
   184	    dom_pending = sum(b["pending"] for b in coh.values())
   185	    red = sum(b["tiers"]["Red"] for b in coh.values())
   186	    yellow = sum(b["tiers"]["Yellow"] for b in coh.values())
   187	    foreign_total = sum(b["foreign"] for b in coh.values())
   188	
   189	    eta_date = None
   190	    eta_days = None
   191	    if dom_pending > 0 and per_day > 0:
   192	        eta_days = -(-dom_pending // per_day)  # ceil
   193	        eta_date = (date.fromisoformat(today) + timedelta(days=eta_days)).isoformat()
   194	
   195	    # Flagged names across cohorts, priority order then Red-before-Yellow.
   196	    flagged = []
   197	    for c in COHORTS:
   198	        reds = [f for f in coh[c]["flagged"] if f["tier"] == "Red"]
   199	        yels = [f for f in coh[c]["flagged"] if f["tier"] == "Yellow"]
   200	        for f in reds + yels:
   201	            flagged.append({**f, "cohort": c})
   202	
   203	    data = {
   204	        "today": today,
   205	        "cycle_start": cycle_start,
   206	        "per_day": per_day,
   207	        "cohorts": coh,
   208	        "totals": {
   209	            "domestic": dom_total, "screened": dom_screened,
   210	            "pending": dom_pending, "foreign": foreign_total,
   211	            "red": red, "yellow": yellow,
   212	            "excluded": sum(b["excluded"] for b in coh.values()),
   213	            "pct": (100.0 * dom_screened / dom_total) if dom_total else 0.0,
   214	        },
   215	        "eta_date": eta_date,
   216	        "eta_days": eta_days,
   217	        "flagged": flagged,
   218	        "generated_at": datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
   219	    }
   220	    data["diff"] = _diff_vs_prior(data)
   221	    return data

   263	def _diff_vs_prior(data: dict) -> dict | None:
   264	    """Compare to the most recent snapshot from a DIFFERENT date."""
   265	    hist = _load_history()
   266	    prior = next((h for h in reversed(hist) if h.get("date") != data["today"]), None)
   267	    if not prior:
   268	        return None
   269	    return {
   270	        "date": prior["date"],
   271	        "screened": data["totals"]["screened"] - prior.get("screened", 0),
   272	        "red": data["totals"]["red"] - prior.get("red", 0),
   273	        "yellow": data["totals"]["yellow"] - prior.get("yellow", 0),
   274	    }

   278	    HEALTH_DIR.mkdir(parents=True, exist_ok=True)

   305	def _cov_table(data: dict) -> list[str]:
   306	    coh = data["cohorts"]
   307	    head = (f"{'RING':<14}{'SCREENED':>10}  {'R / Y':>6}  {'PEND':>5}  "
   308	            f"{'DAYS':>5}  {'DONE BY':>8}  {'GAP':>4}")
   309	    lines = [head, "-" * len(head)]
   310	
   311	    def _row(label, screened, domestic, red, yellow, pending, days, done_by, foreign):
   312	        sc = f"{screened}/{domestic}"
   313	        ry = f"{red} / {yellow}"
   314	        return (f"{label:<14}{sc:>10}  {ry:>6}  {pending:>5}  "
   315	                f"{days:>5}  {done_by:>8}  {foreign:>4}")
   316	
   317	    for c in COHORTS:
   318	        b = coh[c]
   319	        if b["total"] == 0:
   320	            continue
   321	        lines.append(_row(cc.COHORT_LABEL[c], b["screened"], b["domestic"],
   322	                          b["tiers"]["Red"], b["tiers"]["Yellow"], b["pending"],
   323	                          _fmt_days(b.get("eta_days"), b["pending"]),
   324	                          _fmt_done_by(b.get("eta_date"), b["pending"]), b["foreign"]))
   325	    t = data["totals"]
   326	    lines.append("-" * len(head))
   327	    total_days = _fmt_days(data.get("eta_days"), t["pending"])
   328	    total_done = _fmt_done_by(data.get("eta_date"), t["pending"])
   329	    lines.append(_row("TOTAL", t["screened"], t["domestic"], t["red"], t["yellow"],
   330	                      t["pending"], total_days, total_done, t["foreign"]))
   331	    return lines

   521	def post(blocks: list[dict], fallback: str) -> None:
   522	    url = os.environ.get(SLACK_WEBHOOK_ENV, "")
   523	    if not url:
   524	        raise RuntimeError(f"{SLACK_WEBHOOK_ENV} not set — cannot post coverage dashboard")
   525	    payload = json.dumps({"blocks": blocks, "text": fallback}).encode("utf-8")
   526	    req = urllib.request.Request(url, data=payload,
   527	                                 headers={"Content-Type": "application/json"})
   528	    with urllib.request.urlopen(req, timeout=30) as resp:
   529	        if resp.status >= 300:
   530	            raise RuntimeError(f"Slack post failed: HTTP {resp.status}")
